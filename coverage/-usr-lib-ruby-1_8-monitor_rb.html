312 # Monitors provide means of mutual exclusion for Thread programming.
313 # A critical region is created by means of the synchronize method,
314 # which takes a block.
315 # The condition variables (created with #new_cond) may be used 
316 # to control the execution of a monitor with #signal and #wait.
317 #
318 # the Monitor class wraps MonitorMixin, and provides aliases
319 #  alias try_enter try_mon_enter
320 #  alias enter mon_enter
321 #  alias exit mon_exit
322 # to access its methods more concisely.
323 class Monitor
324   include MonitorMixin
325   alias try_enter try_mon_enter
326   alias enter mon_enter
327   alias exit mon_exit
328 end
